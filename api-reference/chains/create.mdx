api: "POST /v1/chain"
---

### Sources

<ResponseField name="id" type="string">
  Unique identifier of this source, if non provided will be auto-generated
</ResponseField>

<ResponseField name="type" type="string">
  Type of the data source (e.g., S3, URL, File, Database)
</ResponseField>

<ResponseField name="config" type="object">
  Configuration settings for the data source
</ResponseField>

<ResponseField name="chunking_strategy" type="Array of Objects">
  Strategies for chunking the data source
</ResponseField>

<Expandable title="chunking_strategy properties">
  <ResponseField name="chunker_type" type="string" optional>
    Type of chunker used (e.g., language.python, markdown, language.html)
  </ResponseField>

  <ResponseField name="chunk_overlap" type="integer">
    Size of overlapping chunks (only applicable for certain types)
  </ResponseField>

  <ResponseField name="chunk_size" type="integer">
    Size of individual chunks
  </ResponseField>

  <ResponseField name="split_at" type="string">
    Point to split at (e.g., heading)
  </ResponseField>
</Expandable>


### Models



<ResponseField name="id" type="string">
  Unique identifier for the model chain
</ResponseField>

<ResponseField name="model" type="string">
  Type of model used (e.g., text_to_text, text_to_image)
</ResponseField>

<ResponseField name="source_id" type="string">
  ID of the associated data source
</ResponseField>

<ResponseField name="config" type="object">
  Configuration settings for the model
</ResponseField>

<ResponseField name="log" type="boolean">
  Whether the model's activities should be logged
</ResponseField>

<Expandable title="config properties">
  <ResponseField name="prompt" type="string">
    Question or instruction for the model to process
  </ResponseField>

  <ResponseField name="system_prompt" type="string">
    Additional directives for model output formatting
  </ResponseField>
</Expandable>



## Response

<ResponseField name="chain_id" type="string" required>
  Chain id
</ResponseField>
